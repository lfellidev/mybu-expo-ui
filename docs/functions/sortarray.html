sortArray

Array sorting utility function

Import

import { sortArray } from 'mybu-expo-ui';

Description

Sorts an array of objects by a specified key. Supports ascending and descending order.

Signature

function sortArray<T>(
  array: T[],
  key: keyof T,
  order?: 'asc' | 'desc'
): T[]

Parameters

Parameter Type Description
array T[] The array to sort
key keyof T The key to sort by
order 'asc' | 'desc' Sort order (default: 'asc')

Returns

T[] - A new sorted array.

Usage Example

import { sortArray } from 'mybu-expo-ui';

const users = [
  { name: 'Charlie', age: 30 },
  { name: 'Alice', age: 25 },
  { name: 'Bob', age: 35 }
];

// Sort by name ascending
sortArray(users, 'name');
// [Alice, Bob, Charlie]

// Sort by age descending
sortArray(users, 'age', 'desc');
// [Bob (35), Charlie (30), Alice (25)]
