getValue

Custom async storage getter

Import

import { getValue } from 'mybu-expo-ui';
Description

Retrieves a value from AsyncStorage. Automatically parses JSON objects.

Signature

async function getValue<T>(
  key: string
): Promise<T | null>

Parameters

Parameter Type Description
key string Storage key to retrieve

Returns

Promise<T | null> - The stored value or null if not found.

Usage Example

import { getValue } from 'mybu-expo-ui';

// Get string
const token = await getValue('token');

// Get object
const user = await getValue('user');
// { name: 'John', id: 1 }

// With type
interface Settings {
  darkMode: boolean;
  notifications: boolean;
}
const settings = await getValue<Settings>('settings');
